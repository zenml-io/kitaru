Access the Kitaru server using one of these methods:

{{- if contains "NodePort" .Values.kitaru.zenml.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Release.Name }}-kitaru)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Kitaru server: http://$NODE_IP:$NODE_PORT"
  kitaru login http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.kitaru.zenml.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        Watch status with: kubectl get svc --namespace {{ .Release.Namespace }} -w {{ .Release.Name }}-kitaru

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Release.Name }}-kitaru --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Kitaru server: http://$SERVICE_IP:{{ .Values.kitaru.zenml.service.port }}"
  kitaru login http://$SERVICE_IP:{{ .Values.kitaru.zenml.service.port }}
{{- else if contains "ClusterIP" .Values.kitaru.zenml.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ .Release.Name }}-kitaru 8080:{{ .Values.kitaru.zenml.service.port }}

  Then connect:

  kitaru login http://localhost:8080
{{- end }}
{{- end }}
